    <script>
        const tg = window.Telegram.WebApp;
        tg.ready(); // сообщаем, что приложение готово
        tg.expand(); // на весь экран

        function createConfetti() {
            for (let i = 0; i < 100; i++) {
                let conf = document.createElement('div');
                conf.classList.add('confetti');
                conf.style.left = Math.random() * 100 + '%';
                conf.style.animationDuration = Math.random() * 1.5 + 1 + 's';
                conf.style.backgroundColor = `hsl(${Math.random() * 360}, 80%, 60%)`;
                document.body.appendChild(conf);
                setTimeout(() => conf.remove(), 2000);
            }
        }

        document.getElementById('joinBtn').addEventListener('click', () => {
            // Отправляем данные боту, что пользователь нажал "Участвовать"
            tg.sendData("join");
            createConfetti();
            const btn = document.getElementById('joinBtn');
            btn.innerText = "✅ Участвую!";
            btn.disabled = true;
            setTimeout(() => {
                tg.close(); // закрыть мини-апп через 1.5 секунды
            }, 1500);
        });
    </script>
